public char repeatedCharacter(String s) {
    boolean[] vis = new boolean[26];
    char c = ' ';
    for (int i = 0; i < s.length(); i++) {
        c = s.charAt(i);
        if (vis[c - 'a']) {
            break;
        } else {
            vis[c - 'a'] = true;
        }
    }
    return c;
}
