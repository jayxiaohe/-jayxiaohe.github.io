/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode removeNthFromEnd(ListNode head, int n) {
        if (head == null || head.next == null) return null;	
		ListNode dummy = new ListNode(0);
		dummy.next = head;
		
		ListNode fast = dummy;
		ListNode slow = dummy;
		
		while (n-- > 0) {
			fast = fast.next;
		}
		while (fast.next != null) {
			fast = fast.next;
			slow = slow.next;
		}
		slow.next = slow.next.next;
		return dummy.next;
    }
}

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
 class Solution {
    public int kthToLast(ListNode head, int k) {
		ListNode slow = head;
		ListNode fast = head;
		
		while (k-- >0) {
			fast = fast.next;
		}
		
		while (fast != null) {
			fast = fast.next;
			slow = slow.next;
		}
		
		return slow.val;
    }
}
